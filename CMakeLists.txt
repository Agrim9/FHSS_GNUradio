########################################################################
# Setup library
########################################################################
include(GrPlatform) #define LIB_SUFFIX

include_directories(${Boost_INCLUDE_DIR})
link_directories(${Boost_LIBRARY_DIRS})

list(APPEND FHSS_sources
${CMAKE_CURRENT_SOURCE_DIR}/blank.cpp
)

add_library(gnuradio-FHSS SHARED ${FHSS_sources})
target_link_libraries(gnuradio-FHSS ${Boost_LIBRARIES} ${GNURADIO_RUNTIME_LIBRARIES})
set_target_properties(gnuradio-FHSS PROPERTIES DEFINE_SYMBOL "gnuradio_FHSS_EXPORTS")

########################################################################
# Install built library files
########################################################################
install(TARGETS gnuradio-FHSS
    LIBRARY DESTINATION lib${LIB_SUFFIX} # .so/.dylib file
    ARCHIVE DESTINATION lib${LIB_SUFFIX} # .lib file
    RUNTIME DESTINATION bin              # .dll file
)

########################################################################
# Build and register unit test
########################################################################
include(GrTest)

include_directories(${CPPUNIT_INCLUDE_DIRS})

list(APPEND test_FHSS_sources
    ${CMAKE_CURRENT_SOURCE_DIR}/test_FHSS.cpp
    ${CMAKE_CURRENT_SOURCE_DIR}/qa_FHSS.cpp
)

add_executable(test-FHSS ${test_FHSS_sources})

target_link_libraries(
  test-FHSS
  ${GNURADIO_RUNTIME_LIBRARIES}
  ${Boost_LIBRARIES}
  ${CPPUNIT_LIBRARIES}
  gnuradio-FHSS
)

GR_ADD_TEST(test_FHSS test-FHSS)
