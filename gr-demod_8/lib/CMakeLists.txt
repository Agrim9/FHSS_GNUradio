########################################################################
# Setup library
########################################################################
include(GrPlatform) #define LIB_SUFFIX

include_directories(${Boost_INCLUDE_DIR})
link_directories(${Boost_LIBRARY_DIRS})

list(APPEND demod_8_sources
)

set(demod_8_sources "${demod_8_sources}" PARENT_SCOPE)
if(NOT demod_8_sources)
	MESSAGE(STATUS "No C++ sources... skipping lib/")
	return()
endif(NOT demod_8_sources)

add_library(gnuradio-demod_8 SHARED ${demod_8_sources})
target_link_libraries(gnuradio-demod_8 ${Boost_LIBRARIES} ${GNURADIO_ALL_LIBRARIES})
set_target_properties(gnuradio-demod_8 PROPERTIES DEFINE_SYMBOL "gnuradio_demod_8_EXPORTS")

if(APPLE)
    set_target_properties(gnuradio-demod_8 PROPERTIES
        INSTALL_NAME_DIR "${CMAKE_INSTALL_PREFIX}/lib"
    )
endif(APPLE)

########################################################################
# Install built library files
########################################################################
include(GrMiscUtils)
GR_LIBRARY_FOO(gnuradio-demod_8 RUNTIME_COMPONENT "demod_8_runtime" DEVEL_COMPONENT "demod_8_devel")

########################################################################
# Build and register unit test
########################################################################
include(GrTest)

include_directories(${CPPUNIT_INCLUDE_DIRS})

list(APPEND test_demod_8_sources
    ${CMAKE_CURRENT_SOURCE_DIR}/test_demod_8.cc
    ${CMAKE_CURRENT_SOURCE_DIR}/qa_demod_8.cc
)

add_executable(test-demod_8 ${test_demod_8_sources})

target_link_libraries(
  test-demod_8
  ${GNURADIO_RUNTIME_LIBRARIES}
  ${Boost_LIBRARIES}
  ${CPPUNIT_LIBRARIES}
  gnuradio-demod_8
)

GR_ADD_TEST(test_demod_8 test-demod_8)

########################################################################
# Print summary
########################################################################
message(STATUS "Using install prefix: ${CMAKE_INSTALL_PREFIX}")
message(STATUS "Building for version: ${VERSION} / ${LIBVER}")
